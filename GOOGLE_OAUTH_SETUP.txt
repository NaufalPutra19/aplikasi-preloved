# GOOGLE OAUTH SETUP - COPY INI KE .env ANDA

# Step 1: Buka https://console.cloud.google.com
# Step 2: Buat Project → Aktifkan Google+ API
# Step 3: Credentials → Create OAuth 2.0 Client ID (Web Application)
# Step 4: Authorized redirect URIs: http://localhost:8000/auth/google/callback
# Step 5: Copy Client ID dan Secret di bawah

# GANTI DENGAN DATA ANDA (Copy dari Google Cloud Console)
GOOGLE_CLIENT_ID=your_client_id_here.apps.googleusercontent.com
GOOGLE_CLIENT_SECRET=your_client_secret_here
GOOGLE_REDIRECT_URI=http://localhost:8000/auth/google/callback
